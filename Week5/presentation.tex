\begin{frame}[fragile]{Referenzen}
\begin{TFCpp}
int i = 1;
int& j = i;
i++; // i = 2
j++; // i = 3
\end{TFCpp}
\end{frame}

\note{
\begin{itemize}
\item Zweiter Name für die gleiche Variable.
\item Eine Referenz wird bei ihrer Deklaration initialisiert!
\item Eine Referenz "zeigt" immer auf die gleiche Variable.
\end{itemize}
}

\begin{frame}[fragile]{Referenzen}
\begin{TFCpp}
void increment (int& m) { //only line changed
m++;
}
int main () {
int n = 3;
increment (n);
return 0;
}
\end{TFCpp}
\end{frame}

\note{
\begin{itemize}
\item Wirkung: m wird neu als Referenz übergeben.
\item Beim Funktionsaufruf wird für n der neue Name m

als Referenz erzeugt.
\item Parameter müssen nicht verändert werden, sie werden

automatisch referenziert.
\item Es ist aber nicht mehr möglich direkt eine Zahl

als Parameter zu übergeben.
\end{itemize}
}
